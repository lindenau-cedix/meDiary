#Beispiele einer Zeile
 (DD.MM(.YYYY)) XX:XX: BB ZZ (AAA), YY ZZ (AAA), YY ZZ (AAA), ...  und YY ZZ (AAA)      <- Aufzählung

 (DD.MM(.YYYY)) XX:XX: BB ZZ (AAA) und YY ZZ (AAA)

 (DD.MM(.YYYY)) XX:XX: BB ZZ (AAA)

 (jetzt:) YY ZZ (AAA) <- wenn keine Zeit angegeben wird aktuelle genommen. Wenn kein Jahr angegeben wird = aktuelles, wenn kein Datum = aktuelles.
 
 
#Legende
 ()=optional; XX=Zeit DD=Tag; MM=Monat; YYYY=Jahr in 4 Ziffern; BB=Substanz; ZZ=Menge; AAA=Notiz
